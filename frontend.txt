step 3:  vite.config.js file
 
import { defineConfig } from "vite";
import react from "@vitejs/plugin-react";

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [react()],
  server: {
    watch: {
      usePolling: true,
    },
    host: true, // needed for the Docker Container port mapping to work
    strictPort: true,
    port: 8080, // you can replace this port with any port
  },
});



step 4: 
docker build . -t shakhawatshihab/frontend_image


step 5: 
docker run -d -it -p 8080:8080 --name frontend_container shakhawatshihab/frontend_image
